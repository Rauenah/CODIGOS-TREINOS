from datetime import datetime, timedelta  # Isso aqui ajuda o computador a entender datas

# Essa fun√ß√£o serve para pegar a data que voc√™ digitou e transformar numa data de verdade
def ler_data(data_digitada):
    formatos = ["%d/%m/%Y", "%d%m%Y"]  # Pode escrever com barra (10/02/2024) ou tudo junto (10022024)

    for formato in formatos:
        try:
            # O computador tenta entender a data de um jeito
            return datetime.strptime(data_digitada, formato)
        except ValueError:
            # Se ele n√£o conseguir entender, tenta o outro jeito
            continue

    # Se ele n√£o conseguir de nenhum jeito, avisa que a data est√° errada
    raise ValueError("Ops! Essa data n√£o existe. Tente de novo!")

# Essa fun√ß√£o descobre quando voc√™ est√° no per√≠odo f√©rtil
def calcular_periodo_fertil(data_menstruacao, ciclo=28):
    # Primeiro, o computador descobre qual foi o dia da sua √∫ltima menstrua√ß√£o
    data_inicio = ler_data(data_menstruacao)

    # A ovula√ß√£o acontece mais ou menos 14 dias antes do pr√≥ximo ciclo
    # Ent√£o a gente faz uma continha: come√ßo + (ciclo - 14)
    dia_ovulacao = data_inicio + timedelta(days=ciclo - 14)

    # O per√≠odo f√©rtil n√£o √© s√≥ um dia.
    # Ele √© 3 dias antes e 3 dias depois da ovula√ß√£o.
    inicio = dia_ovulacao - timedelta(days=3)
    fim = dia_ovulacao + timedelta(days=3)

    # Aqui retornamos o come√ßo e o fim desses dias especiais
    return inicio, fim

# Agora come√ßa o programa de verdade
print("=== Calculadora de Per√≠odo F√©rtil ===")

# Aqui o computador pergunta qual foi o dia da sua √∫ltima menstrua√ß√£o
data = input("Digite o dia que sua √∫ltima menstrua√ß√£o come√ßou: ")

# Aqui ele pergunta quantos dias dura seu ciclo (se voc√™ apertar ENTER, ele usa 28)
ciclo = int(input("Quantos dias dura seu ciclo? (se n√£o souber, apenas aperte ENTER): ") or 28)

# Aqui ele descobre o per√≠odo f√©rtil usando a fun√ß√£o l√° de cima
inicio, fim = calcular_periodo_fertil(data, ciclo)

# E agora ele mostra as datas bonitinhas
print(f"Seu per√≠odo f√©rtil vai de {inicio.strftime('%d/%m/%Y')} at√© {fim.strftime('%d/%m/%Y')}")
